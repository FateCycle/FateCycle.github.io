线程协作

  • Lock

    • 特点:

      比起synchronized,实现了更加复杂的临界区的结构

      tryLock方法可以预判是否空闲

      允许分离读写操作，多个读，一个写

      性能更好

    • ReentrantLock类，可重入的互斥锁

    • ReentrantReadWriteLock类，可重入的读写锁

    • lock和unlock函数

    public class LockExample {
    // 可重入的互斥锁
    private static final ReentrantLock queryLock = new ReentrantLock();

    //可重入的读写锁
    private static final ReentrantReadWriteLock orderLock = new ReentrantReadWriteLock();

    public static void main(String[] args) throws InterruptedException {

         // handleOrder();

         buyMilkTea();

    }


  public static void handleOrder() throws InterruptedException {
      LockExample locakExample = new LockExample();
      Thread boss = new Thread(new Runnable() {

          @Override
          public void run() {
              while(true) {
                  try {
                      locakExample.addOrder();
                      long waitingTime = (long)(Math.random()*1000);
                      Thread.sleep(waitingTime);
                  }catch (Exception e) {
                      e.printStackTrace();
                  }
              }

          }
      });
      boss.start();

      int workerCnt = 3;
      Thread[] workers = new Thread[workerCnt];
      for(int i=0;i<workerCnt;i++) {
          workers[i] = new Thread(new Runnable() {

              @Override
              public void run() {
                  while(true) {
                      try {
                          locakExample.viewOrder();
                          long workingTime = (long)(Math.random()*5000);
                          Thread.sleep(workingTime);

                      } catch (Exception e) {
                          e.printStackTrace();
                      }
                  }

              }
          });
          workers[i].start();
      }
  }

  public void addOrder() throws InterruptedException {
      // 写者锁只支持单个线程访问关键区
      orderLock.writeLock().lock();
      long writingTime = (long)(Math.random()*1000);
      Thread.sleep(writingTime);
      System.out.println("老板新家一笔订单");
      orderLock.writeLock().unlock(); 
  }

  public void viewOrder() throws InterruptedException {
      // 读者锁支持多个线程访问关键区
      orderLock.readLock().lock();

      long readTime = (long)(Math.random()*500);
      Thread.sleep(readTime);
      System.out.println(Thread.currentThread().getName()+":查案订单本");
      orderLock.readLock().unlock();
  }

  public static void buyMilkTea() throws InterruptedException {
      LockExample lockExample = new LockExample();

          if(queryLock.tryLock()) {

              System.out.println(Thread.currentThread().getName()+"再等等");
          }
          if(flag) {
              Thread.sleep(1000);

      }

+ Semaphore

  信号量:本质是个计数器，计数器大于0可以使用,等于0不能使用,使用信号量可以实现多个并发量，他可以比Lock做到更细粒度的控制，可以控制多个线程同时访问关键区

  API: acquire 获取以及release 释放

  ```java
public class SemaphoreExample {

    private final Semaphore placeSemaphore = new Semaphore(5);

      public boolean parking() throws InterruptedException {
          //判断信号量是否是0
          if(placeSemaphore.tryAcquire()) {
              System.out.println(Thread.currentThread().getName()+":停车成功");
              return true;
          } else {
              System.out.println(Thread.currentThread().getName()+":停车失败");
              return false;
          }
      }

      public void leaving() throws InterruptedException {
          placeSemaphore.release();
          System.out.println(Thread.currentThread().getName()+":开车");
      }

      public static void main(String[] args) throws InterruptedException {
          int tryToParkCnt = 10;

          SemaphoreExample semaphoreExample = new SemaphoreExample();
          Thread[] parkers = new Thread[tryToParkCnt];

          for(int i=0;i<tryToParkCnt;i++) {
              parkers[i] = new Thread(new Runnable() {

                  @Override
                  public void run() {
                      try {
                          long randomTime = (long) (Math.random()*1000);
                          Thread.sleep(randomTime);
                          if(semaphoreExample.parking()) {
                              long radomTime = (long)(Math.random()*1000);
                              Thread.sleep(randomTime);
                              semaphoreExample.leaving();
                          }
                      }catch(Exception e) {
                              e.printStackTrace();
                      }
                  }
              });

              parkers[i].start();
          }


          for(int i=0;i<tryToParkCnt;i++) {
              // main函数需要等待所有parkers[i]结束才会继续执行
              parkers[i].join();
          }
      }


  }

  • Latch

    等待锁，一个同步辅助类，用来同步执行任务的一个或者多个线程

    不是用来保护临界区或者共享资源

    CountDownLatch() 计数减一

    await() 等待latch变成0才执行不然阻塞

    public class CountDownLatchExample {

    public static void main(String[] args) throws InterruptedException {
        int runnerCnt = 10;
        CountDownLatch startSignal = new CountDownLatch(1);
        CountDownLatch downSingnal = new CountDownLatch(runnerCnt);

        for(int i=0;i<runnerCnt;i++) {
            new Thread(new Worker(startSignal,downSingnal)).start();
        }
        System.out.println("准备工作");
        System.out.println("准备工作就绪");
        startSignal.countDown();
        System.out.println("比赛开始");
        downSingnal.await();  //等待downSingnal由10变成0,代表全部跑完
        System.out.println("比赛结束");

    }

    static class Worker implements Runnable {
        private final CountDownLatch startSingal;

        private final CountDownLatch downSingal;



      public Worker(CountDownLatch startSingal, CountDownLatch downSingal) {
          super();
          this.startSingal = startSingal;
          this.downSingal = downSingal;
      }



      @Override
      public void run() {
          // TODO Auto-generated method stub
          try {
              startSingal.await();  //等待startSingal从1为0，代表开始赛跑
              doWork();
              downSingal.countDown();
          } catch (Exception e) {
              // TODO: handle exception
              e.printStackTrace();
          }

      }

      void doWork() {
          System.out.println(Thread.currentThread().getName()+":跑完全屏");
      }
  }

}




+ Barrier

  集合点，一个同步辅助类

  允许多个线程在同一个点上工作

  实现类为CyclicBarrier(构造函数是传入同步线程数量，await是等待其他线程)，到达数量后就放行

  ```java
  public class CyclicBarrierExample {

      public static void main(String[] args) {
          final int[][] number = new int[3][5];
          final int[] results = new int[3];
          int[] result1 = new int[] {1,2,3,4,5};
          int[] result2 = new int[] {6,7,8,9,10};
          int[] result3 = new int[] {11,12,13,14,15};
          number[0] = result1;
          number[1] = result2;
          number[2] = result3;


          CaculateFinalResult caculateFinalResult = new CaculateFinalResult(results);
          //出现三个线程运行caculateFinalResult线程(即run方法进行求和)
          CyclicBarrier cyclicBarrier = new CyclicBarrier(3,caculateFinalResult);

          for(int i=0;i<3;i++) {
              // 这边传入的barrier表示等待出现三个calculateEachRow线程
              CalculateEachRow calculateEachRow = new CalculateEachRow(number, i, results, cyclicBarrier);
              //开启行统计线程
              new Thread(calculateEachRow).start();
          }

      }

  }


  class CalculateEachRow implements Runnable {
      final int[][] number;
      final int rowNumber;
      final int[] res;
      final CyclicBarrier barrier;
      public CalculateEachRow(int[][] number, int rowNumber, int[] res, CyclicBarrier barrier) {
          super();
          this.number = number;
          this.rowNumber = rowNumber;
          this.res = res;
          this.barrier = barrier;
      }
      @Override
      public void run() {
          int[] row = number[rowNumber];
          int sum = 0;
          for(int data: row) {
              sum+=data;
              res[rowNumber] = sum;
          }

          try {
              System.out.println(Thread.currentThread().getName()+":计算器"+(rowNumber+1)+"行结果，结果为: "+sum);
              //等待出现三个线程，barrier设置的是三个回调设置的是CaculateFinalResult线程
              barrier.await();
          }catch (InterruptedException | BrokenBarrierException e) {
              e.printStackTrace();

          }

      };
  }


  class CaculateFinalResult implements Runnable {

      final int[] eachRowRes;
      int finalRes;

      public CaculateFinalResult(int[] eachRowRes) {
          this.eachRowRes = eachRowRes;
      }

      public int getFinalRes() {
          return finalRes;
      }

      public void setFinalRes(int finalRes) {
          this.finalRes = finalRes;
      }

      @Override
      public void run() {
          int sum = 0;
          for(int data: eachRowRes) {
              sum+=data;
          }
          finalRes = sum;
          System.out.println("最终结果为"+ finalRes);
      }

  }

  • Phaser

    允许执行并发多阶段任务，同步辅助类

    当每个阶段结束的位置对线程同步，当所有线程都到达这一步，在进行下一步

    方法:arrive(),arriveAndAwaitAdvance()

    public class PhaserExample {


  public static void main(String[] args) {
      int studentsCnt = 5;
      // 设置同时开始的线程个数
      Phaser phaser = new Phaser(studentsCnt);
      for(int i=0;i<studentsCnt ;i++ ) {
          new Thread(new Student(phaser)).start();
      }
  }

}

class Student implements Runnable {

  private final Phaser phaser;

  public Student(Phaser phaser) {
      super();
      this.phaser = phaser;
  }

  @Override
  public void run() {
      try {
          doTesting(1);
          // 等到5线程完成才能进行下一个任务
          phaser.arriveAndAwaitAdvance();
          doTesting(2);
          phaser.arriveAndAwaitAdvance();
          doTesting(3);
          phaser.arriveAndAwaitAdvance();
      } catch(Exception e) {
          e.printStackTrace();
      }
  }

  private void doTesting(int i) throws InterruptedException {
      String name = Thread.currentThread().getName();
      System.out.println(name+"开始答题"+"i"+"题");
      long thinkTime = (long)(Math.random()*1000);
      Thread.sleep(thinkTime);
      System.out.println(name+"第"+i+"道题答题结束");
  }

}


+ Exchanger

  允许并发线程中相互交换消息

  允许两个线程中定义同步点，当两个线程到达同步点，他们交换数据结构

  方法: exchange() 线程双方交换数据，交换数据是双向的

```java
public class ExchangerExample {

    public static void main(String[] args) throws InterruptedException {
        // 交换的数据类型为String
        Exchanger<String> exchanger = new Exchanger<String>();
        BackgroundWorker worker = new BackgroundWorker(exchanger);
        new Thread(worker).start();
        Scanner scanner = new Scanner(System.in);
        while(true) {
            System.out.println("输入想要查询的属性");
            // 截取到回车
            String input = scanner.nextLine().trim();
            // 对应子线程的String item = exchanger.exchange(null); 
            exchanger.exchange(input);
            String value = exchanger.exchange(null);
            if("exit".equals(value)) {
                break;
            }
            System.out.println("查询结果"+value);
        }
        scanner.close();


    }

}

class BackgroundWorker implements  Runnable {
    final Exchanger<String> exchanger ;

    public BackgroundWorker(Exchanger<String> exchanger) {
        super();
        this.exchanger = exchanger;
    }


    public void run() {
        while(true) {
            try {
                // 线程遇到exchange会等待另一个线程的exchange然后把括号中的变量相互交换给对方，返回值即是即交换结果
                //当然你可以不选择接收
                String item = exchanger.exchange(null); 
                switch (item) {
                case "zhangsan":
                    exchanger.exchange("90");
                    break;
                case "lisi":
                    exchanger.exchange("80");
                    break;
                case "wangwu":
                    exchanger.exchange("70");
                    break;
                case "exit":
                    exchanger.exchange("exit");
                    break;

                default:
                    exchanger.exchange("查无此人");
                    break;
                }
            }catch(Exception e) {
                e.printStackTrace();
            }
        }

    }
}
