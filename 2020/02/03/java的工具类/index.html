工具类(针对数组和集合的一些操作类)

  • Arrays(针对数组)

    排序: sort/paralleSort

    查找: binarySearch

    批量拷贝: copyOf

    批量赋值: fill

    等价性: equals

  • Collections(针对Collection及其子类)

    排序: sort/paralleSort

    查找: binarySearch

    批量赋值: fill

    最大,最小: max,min

    反序:reverse

示例代码:

// 比较器要么额外声明类并实现Comparator要么在Person内部实现Comparable接口
package xXX;

import java.util.Comparator;

public class Person2Compare implements Comparator<Person> {

    @Override
    public int compare(Person arg0, Person arg1) {
        if (arg0.getAge() == arg1.getAge()) {
            return arg0.getName().compareTo(arg1.getName());
        } else {
            return arg0.getAge() - arg1.getAge();
        }

    }
}


public class CollectionTest {

    public static void main(String args[]) {
        ArrayList<Person> list  = new ArrayList<Person>();
        list.add(new Person("UUU",16));
        list.add(new Person("BBB",19));
        list.add(new Person("AAA",19));
        System.out.println(Collections.binarySearch(list, new Person("BBB", 19),new Person2Compare()));
        Collections.sort(list,new Person2Compare());
        for(Person item : list) {
            System.out.println(item.getName()+" : "+item.getAge());
        }
        System.out.println(Collections.max(list, new Person2Compare()));
        System.out.println(Collections.min(list,new Person2Compare()));

        int[] arr1 = new int[3];
        Arrays.fill(arr1,0,3, 8);  // [0,3)赋值为8
        int[] arr2 = Arrays.copyOf(arr1, 2);
        System.out.println(Arrays.toString(arr1));
        System.out.println(Arrays.toString(arr2));
        System.out.println(Arrays.equals(arr1, arr2));
    }

}
