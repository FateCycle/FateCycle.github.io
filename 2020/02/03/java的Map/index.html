Map

Hashtable 同步，慢，数据量小

HashMap 不同步，快，数据量大，不会排序

LinkedHashMap 按照插入的先后排序

TreMap 按照key的自然排序或者compareTo 方法进行排序

Properties 同步，文件形式，数据量小

Hashtable主要的api:

  • clear 清空

  • contains 等价于containsValue

  • containsValue 是否包含某个值

  • containsKey 是否包含某个值

  • get 获取特定键的值

  • put 放入键值对

  • remove 通过键移除键值对

  • size 得到元素个数

Properties 除了以上的方法还有

  • store
  • load
  • getProperty
  • setProperty

下面为Map的一些示例代码


public class HashMapTest {
    public static void main(String args[]) {
        Hashtable<Integer,String> ht = new Hashtable<Integer,String>();
        ht.put(1, "haha");
        ht.put(2, "heihei");
        ht.put(1, "UUU");
        Iterator<Entry<Integer,String>> ite = ht.entrySet().iterator();
        while(ite.hasNext()) {
            Map.Entry<Integer,String> entry = ite.next();
            System.out.println(entry.getKey()+"-->"+entry.getValue());
        }

        Iterator<Integer> ite1 = ht.keySet().iterator();
        while(ite1.hasNext()) {
            Integer key = ite1.next();
            System.out.println(key+"-->"+ht.get(key));
        }
        // Hashtable专属，HashMap，LinkedHashMap都不能使用
        Enumeration<Integer> keys = ht.keys();
        while(keys.hasMoreElements()) {
            Integer key = keys.nextElement();
            System.out.println(key+"-->"+ht.get(key));
        }
    }

}

//properties文件示例
public class PropertiesTest {

    public static String GetValueByKey(String filePath,String key) {
        Properties pps = new Properties();
        try {
            InputStream in = new BufferedInputStream(new FileInputStream(filePath));
            pps.load(in);
            String value = pps.getProperty(key);
            return value;
        }catch(IOException e) {
            e.printStackTrace();
            return null;
        }
    }

    public static void GetAllProperties(String filePath) throws IOException {
        Properties pps = new Properties();
        InputStream in = new BufferedInputStream(new FileInputStream(filePath));
        pps.load(in);
        Enumeration<?> en = pps.propertyNames();
        while (en.hasMoreElements()) {
            String strKey = (String) en.nextElement();
            String strValue = pps.getProperty(strKey);
        }
    }

    public static void WriteProperties(String filePath,String pkey,String pvalue) throws IOException{
        File file = new File(filePath);
        if(!file.exists()) {
            file.createNewFile();
        }

        Properties pps = new Properties();
        InputStream in  = new FileInputStream(filePath);
        pps.load(in);
        OutputStream out =new FileOutputStream(filePath);
        pps.setProperty(pkey, pvalue);
        pps.store(out, "Update" + pkey + "name");
        out.close();
    }

    public static void main(String[] args) throws IOException {
        WriteProperties("Test.properties","name","12345");
        GetAllProperties("Test.properties");
        String value = GetValueByKey("Test.properties","name");
        System.out.println("name:"+value);

    }

}
