文件读写

文件系统是由操作系统来管理的，他和java进程是平行的，文件分隔符(linux/unix使用/分隔,window使用双反斜杠来分隔)

文件整体属性

java.io.File来进行文件操作

常用方法:

createNewFile,delete,exists,getAbsolutePath,getName,getParent,getPath,isDirectory,isFile,length,listFile,mkdir,mkdirs

File不涉及文件内容只涉及文件属性

// 示例代码
File dir =new File("D:/file");
        if (!dir.exists()) {
            dir.mkdirs();
        }

        System.out.println(dir.isDirectory());
        File file = new File("D:/file/abc.txt");
        if (!file.exists()) {
            try {
                file.createNewFile();
            }catch(IOException e) {
                e.printStackTrace();
            }

        System.out.println(file.isFile());
        System.out.println(file.getName());
        System.out.println(file.getParent());

        System.out.println(file.length());
        System.out.println(file.lastModified());

        //不包含子文件夹
        File[] filelist = dir.listFiles();
        for(File f : filelist) {
            System.out.println(file.getPath());
        }
    }

path,File,DirectoryStream,FileVisitor,FileSystem使得可以实现文件的复制和移动，文件相对路径，递归遍历目录，递归删除目录等

File类API:

noExists 测试所在文件是否存在

isDirectory 判断是否是目录

createDirectories 创建所有存在的父目录在创建当前需要目录,即:可以创建多级目录

createFile 创建文件

newDirectoryStream 打开一个目录

readAttributes 读取属性

public class FilesTet {

    // 移动文件
    public static void moveFile() {
        // 将C:/temp/abc.txt 移动到C:/temp/test下并改名为abxxopy.txt
        Path from = Paths.get("D:/file", "abc.txt");
        Path to = from.getParent().resolve("test/abccopy.txt");
        try {
            System.out.println(Files.size(from));
            Files.move(from,to,StandardCopyOption.REPLACE_EXISTING);
        } catch(IOException e) {
            System.out.println("移动文件错误");
        }
    }

    //  读取文件属性
    public static void fileAttributes() {
        Path path = Paths.get("D:/file");
        System.out.println(Files.isDirectory(path, LinkOption.NOFOLLOW_LINKS));
        try {
            BasicFileAttributes attributes = Files.readAttributes(path, BasicFileAttributes.class);
            System.out.println(attributes.lastModifiedTime().toMillis());
        }catch(IOException e) {
            e.printStackTrace();
        }
    }


    public static void createDirectory() {
        Path path = Paths.get("D:/file/test");
        try {
            if(Files.notExists(path)) {
                Files.createDirectories(path);
            }
            // 路径拼接
            Path path2 = path.resolve("A.java");
            Files.createFile(path2);
            // 打开目录并挑选后缀为java的文件
            DirectoryStream<Path> paths = Files.newDirectoryStream(path,"*.{java,txt}");
            for(Path p : paths) {
                System.out.println(p.getFileName());
            }
        }catch(IOException e) {
            e.printStackTrace();
        }

    }


    public static void main(String args[]) {
        moveFile();
        fileAttributes();
        createDirectory();
    }
}

FileVisitor类API:

walkFileTree 递归访问文件树的每一个文件

这个API需要一个实现FileVisitor接口的查找类，具体事例:

import java.io.IOException;
import java.nio.file.FileSystems;
import java.nio.file.FileVisitOption;
import java.nio.file.FileVisitResult;
import java.nio.file.FileVisitor;
import java.nio.file.Files;
import java.nio.file.Path;
import java.nio.file.PathMatcher;
import java.nio.file.Paths;
import java.nio.file.attribute.BasicFileAttributes;
import java.util.EnumSet;

class Search implements FileVisitor{

    private final PathMatcher matcher;

    public Search(String ext) {
        matcher = FileSystems.getDefault().getPathMatcher("glob:" + ext);
    }

    public void JudgeFile(Path file) throws IOException {
        Path name = file.getFileName();
        if (name!=null&&matcher.matches(name)) {
            System.out.println("找到了"+name);
        }
    }

    @Override
    public FileVisitResult postVisitDirectory(Object arg0, IOException arg1) throws IOException {
        // TODO Auto-generated method stub
        return FileVisitResult.CONTINUE;
    }

    @Override
    public FileVisitResult preVisitDirectory(Object arg0, BasicFileAttributes arg1) throws IOException {

        return FileVisitResult.CONTINUE;
    }

    @Override
    public FileVisitResult visitFile(Object arg0, BasicFileAttributes arg1) throws IOException {
        // TODO Auto-generated method stub
        JudgeFile((Path)arg0);
        return FileVisitResult.CONTINUE;
    }

    @Override
    public FileVisitResult visitFileFailed(Object arg0, IOException arg1) throws IOException {
        // TODO Auto-generated method stub
        return FileVisitResult.CONTINUE;
    }


}
public class SearchJAVAFiles {
    public static void main(String args[]) throws IOException {
        String ext = "*.java";
        Path fileTree = Paths.get("D:/file");
        Search walk = new Search(ext);
        EnumSet<FileVisitOption> opts = EnumSet.of(FileVisitOption.FOLLOW_LINKS);

        // 递归查找子文件
        Files.walkFileTree(fileTree, opts, Integer.MAX_VALUE, walk);

    }
}
